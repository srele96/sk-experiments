coordinate:
  private:
    k
    j

    Up():
      --k
    Down():
      ++k
    Left():
      --j
    Right():
      ++j

    UpRight():
      Up()
      Right()
    DownLeft():
      Down()
      Left()

  public:
    step(Direction direction):
      direction.UP:
        Up()
      direction.DOWN:
        Down()
      direction.LEFT:
        Left()
      direction.RIGHT:
        Right()
      direction.UP_RIGHT:
        UpRight()
      direction.DOWN_LEFT:
        DownLeft()
